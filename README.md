# Comparison Operators

< ---> Strictly less than
<= ---> Less than or equal
> ---> Strictly greater than
>= ---> Greater than or equal
== ---> Equal
!= ---> Not equal

print(bmi) ---> array([21.852, 20.975, 21.75, 24.747, 21.441])
print(bmi>23) ---> array([False, False, False, True, False])
print(bmi[bmi>23]) ---> array([24.747])
